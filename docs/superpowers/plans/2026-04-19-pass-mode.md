# Pass Mode Toggle Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a segmented pill to the Job section that lets the user choose between specifying total depth (inches) or number of passes (integer) to drive toolpath depth pass count.

**Architecture:** `passMode` and `numPasses` are added to `SurfacingParams`. `toolpath.ts` gets a `resolveDepthPassCount()` helper that returns the correct pass count for whichever mode is active. The UI replaces the "Total Depth" label with an inline pill; a hidden `<input id="passMode">` tracks the active mode; `getFormValues` interprets the shared depth input based on that mode.

**Tech Stack:** TypeScript, Bun test runner (`bun:test`), happy-dom for DOM tests, Vite + vite-plugin-singlefile

export function resolveDepthPassCount(params: SurfacingParams): number {
  if (params.passMode === 'numPasses') return params.numPasses
  if (params.totalDepth <= 0) return 0
  return Math.ceil(params.totalDepth / params.depthPerPass - 1e-9)
}

// Wire up pass-mode pill
const pillButtons = form.querySelectorAll('.pass-mode-btn')
const depthInput = form.querySelector('#depthInput') as HTMLInputElement
const depthUnit = form.querySelector('#depthUnit') as HTMLSpanElement
const passModeInput = form.querySelector('#passMode') as HTMLInputElement

pillButtons.forEach(btn => {
  btn.addEventListener('click', () => {
    const mode = (btn as HTMLElement).dataset.mode as 'totalDepth' | 'numPasses'
    pillButtons.forEach(b => b.classList.remove('active'))
    btn.classList.add('active')
    passModeInput.value = mode
    depthInput.value = ''
    if (mode === 'totalDepth') {
      depthInput.inputMode = 'decimal'
      depthInput.pattern = '[0-9]+(\\.[0-9]+)?'
      depthInput.dataset.tooltip = 'Total material to remove. Leave blank for skim-only run.'
      depthUnit.textContent = 'in'
    } else {
      depthInput.inputMode = 'numeric'
      depthInput.pattern = '[0-9]+'
      depthInput.dataset.tooltip = 'Number of depth passes to make.'
      depthUnit.textContent = ''
    }
    onUpdate(getFormValues(form))
  })
})
